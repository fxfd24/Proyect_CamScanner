 * Endpoints funcionando:
 * - /        → Página HTML
 * - /capture → Captura y envía foto JPG
 */

#include "esp_camera.h"
#include <WiFi.h>

// ===== CONFIGURACIÓN WiFi =====
const char* ssid = "TU_WIFI_SSID";
const char* password = "TU_WIFI_PASSWORD";

// ===== PINES PARA AI-THINKER ESP32-CAM =====
#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22

// LED Flash (GPIO 4)
#define LED_FLASH 4

// Servidor web
WiFiServer server(80);

// Contador de fotos
int foto_count = 0;

void setup() {
  Serial.begin(115200);
  Serial.println("\n\n=== INICIANDO ESP32-CAM ===");
  
  // Configurar LED flash
  pinMode(LED_FLASH, OUTPUT);
  digitalWrite(LED_FLASH, LOW);
  
  // Inicializar cámara
  Serial.println("Inicializando cámara...");
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;
  config.frame_size = FRAMESIZE_SVGA;  // 800x600
  config.jpeg_quality = 12;
  config.fb_count = 1;

  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("Error cámara: 0x%x\n", err);
    return;
  }
  Serial.println("✅ Cámara lista");

  // Conectar WiFi
  Serial.printf("Conectando a WiFi: %s", ssid);
  WiFi.begin(ssid, password);
  
  int intentos = 0;
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    intentos++;
    if (intentos > 40) {
      Serial.println("\n❌ No se pudo conectar a WiFi");
      return;
    }
  }
  
  Serial.println("\n✅ WiFi conectado!");
  Serial.print("📡 IP: http://");
  Serial.println(WiFi.localIP());
  
  // Iniciar servidor
  server.begin();
  Serial.println("🚀 Servidor web iniciado");
  Serial.println("   • /        - Página HTML");
  Serial.println("   • /capture - Capturar foto (FLASH + JPG)");
  Serial.println("========================");
}

void loop() {
  WiFiClient client = server.available();
  
  if (client) {
    String request = client.readStringUntil('\r');
    client.flush();
    
    // Mostrar petición en serial
    Serial.println("\n📱 Petición recibida:");
    Serial.println("   " + request.substring(0, 50) + "...");
    
    // ===== ENDPOINT /capture =====
    if (request.indexOf("GET /capture") != -1) {
      foto_count++;
      Serial.printf("📸 Capturando foto #%d...\n", foto_count);
      
      // Encender flash
      digitalWrite(LED_FLASH, HIGH);
      delay(100);
      
      // Capturar imagen
      camera_fb_t * fb = esp_camera_fb_get();
      if (!fb) {
        Serial.println("❌ Error capturando");
        client.println("HTTP/1.1 500 Internal Server Error");
        client.println("Content-Type: text/plain");
        client.println("Connection: close");
        client.println();
        client.println("Camera capture failed");
        digitalWrite(LED_FLASH, LOW);
        client.stop();
        return;
      }
      
      // Apagar flash
      digitalWrite(LED_FLASH, LOW);
      
      // Enviar headers HTTP
      client.println("HTTP/1.1 200 OK");
      client.println("Content-Type: image/jpeg");
      client.println("Content-Length: " + String(fb->len));
      client.println("Access-Control-Allow-Origin: *");  // Importante para Python
      client.println("Connection: close");
      client.println();
      
      // Enviar imagen
      client.write(fb->buf, fb->len);
      
      Serial.printf("✅ Foto enviada: %d bytes\n", fb->len);
      
      // Liberar memoria
      esp_camera_fb_return(fb);
    }
    
    // ===== ENDPOINT / (página principal) =====
    else if (request.indexOf("GET / ") != -1 || request.indexOf("GET / HTTP") != -1) {
      client.println("HTTP/1.1 200 OK");
      client.println("Content-Type: text/html");
      client.println("Connection: close");
      client.println();
      client.println("<!DOCTYPE html>");
      client.println("<html>");
      client.println("<head><title>ESP32-CAM</title></head>");
      client.println("<body>");
      client.println("<h1>📸 ESP32-CAM</h1>");
      client.println("<p>IP: " + WiFi.localIP().toString() + "</p>");
      client.println("<p>Fotos tomadas: " + String(foto_count) + "</p>");
      client.println("<p><a href='/capture' target='_blank'>📸 TOMAR FOTO</a></p>");
      client.println("</body>");
      client.println("</html>");
    }
    
    // ===== OTROS ENDPOINTS =====
    else {
      client.println("HTTP/1.1 404 Not Found");
      client.println("Content-Type: text/plain");
      client.println("Connection: close");
      client.println();
      client.println("404 - Endpoint no encontrado");
    }
    
    client.stop();
  }
  
  delay(10);
}

// ============================================
// PARTE 1: STREAMING MJPEG
// ============================================
static esp_err_t stream_handler(httpd_req_t *req) {
  camera_fb_t *fb = NULL;
  esp_err_t res = ESP_OK;
  size_t _jpg_buf_len = 0;
  uint8_t *_jpg_buf = NULL;
  char part_buf[64];
  
  res = httpd_resp_set_type(req, "multipart/x-mixed-replace;boundary=123456789000000000000987654321");
  if (res != ESP_OK) return res;
  
  Serial.println("🎥 Streaming iniciado");
  
  while (true) {
    fb = esp_camera_fb_get();
    if (!fb) {
      res = ESP_FAIL;
      break;
    }
    
    if (fb->format != PIXFORMAT_JPEG) {
      bool jpeg_converted = frame2jpg(fb, 80, &_jpg_buf, &_jpg_buf_len);
      esp_camera_fb_return(fb);
      fb = NULL;
      if (!jpeg_converted) continue;
    } else {
      _jpg_buf_len = fb->len;
      _jpg_buf = fb->buf;
    }
    
    size_t hlen = snprintf(part_buf, 64, "Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n", _jpg_buf_len);
    res = httpd_resp_send_chunk(req, part_buf, hlen);
    if (res == ESP_OK) res = httpd_resp_send_chunk(req, (const char*)_jpg_buf, _jpg_buf_len);
    if (res == ESP_OK) res = httpd_resp_send_chunk(req, "\r\n--123456789000000000000987654321\r\n", 37);
    
    if (fb) {
      esp_camera_fb_return(fb);
      fb = NULL;
      _jpg_buf = NULL;
    } else if (_jpg_buf) {
      free(_jpg_buf);
      _jpg_buf = NULL;
    }
    
    if (res != ESP_OK) break;
  }
  
  Serial.println("🎥 Streaming finalizado");
  return res;
}

// ============================================
// PARTE 2: CAPTURA DE FOTO
// ============================================
static esp_err_t capture_handler(httpd_req_t *req) {
  foto_count++;
  Serial.printf("\n📸 Capturando foto #%d...\n", foto_count);
  
  // Flash
  digitalWrite(LED_FLASH, HIGH);
  delay(100);
  
  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) {
    digitalWrite(LED_FLASH, LOW);
    httpd_resp_send_500(req);
    return ESP_FAIL;
  }
  
  digitalWrite(LED_FLASH, LOW);
  
  // Headers
  httpd_resp_set_type(req, "image/jpeg");
  httpd_resp_set_hdr(req, "Access-Control-Allow-Origin", "*");
  httpd_resp_set_hdr(req, "Content-Disposition", "inline; filename=captura.jpg");
  
  esp_err_t res = httpd_resp_send(req, (const char*)fb->buf, fb->len);
  
  Serial.printf("✅ Foto enviada: %d bytes\n", fb->len);
  esp_camera_fb_return(fb);
  return res;
}

static esp_err_t index_handler(httpd_req_t *req) {
  char html[1024];
  snprintf(html, sizeof(html),
    "<!DOCTYPE html>"
    "<html><head><title>ESP32-CAM</title>"
    "<meta name='viewport' content='width=device-width, initial-scale=1'>"
    "<style>body{font-family:Arial;text-align:center;margin:20px;background:#f0f0f0}"
    ".container{background:white;padding:20px;border-radius:10px}"
    "img{max-width:100%%;border:2px solid #333;border-radius:5px}"
    "button{background:#4CAF50;color:white;padding:15px 30px;font-size:18px;border:none;border-radius:5px;cursor:pointer}"
    "button:hover{background:#45a049}</style>"
    "</head><body><div class='container'>"
    "<h1>📸 ESP32-CAM</h1>"
    "<p><b>IP:</b> %s</p>"
    "<p><b>Fotos:</b> %d</p>"
    "<img src='/stream' width='640' height='480'/>"
    "<p><button onclick='fetch(\"/capture\").then(()=>alert(\"Foto capturada!\"))'>📸 TOMAR FOTO</button></p>"
    "</div></body></html>",
    WiFi.localIP().toString().c_str(), foto_count);
  
  httpd_resp_set_type(req, "text/html");
  httpd_resp_send(req, html, strlen(html));
  return ESP_OK;
}
